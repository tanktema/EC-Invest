country_event = {
	id = ECInvest.3 
	
	hidden = yes 
	is_triggered_only = yes 
	
	immediate = {
		FROM.FROM = {
			add_building_construction = { 
				type = eco_power_plant
				level = 1
				instant_build = yes 
			}
			owner = {
				ECI_validate_country_value_arrays = yes 
			}
		}
		FROM = {
			if = {
				limit = {
					FROM.FROM =  { check_variable = { eco_power_plant < 4 } }
				}
				add_to_variable = { budget_reserve = 2 } 
			}
			if = {
				limit = {
					FROM.FROM =  { check_variable = { eco_power_plant < 6 } }
				}
				add_to_variable = { budget_reserve = 7 } 
			}
			if = {
				limit = {
					FROM.FROM =  { check_variable = { eco_power_plant > 5 } }
				}
				add_to_variable = { budget_reserve = 12 } 
			}
		}
	}
}
